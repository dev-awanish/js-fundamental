  const at=arr.reduce((arr,obj)=> {
     if(input > obj) {
      if(!matched && !(input % obj) && input/obj && input/obj !==1)
        {
        	matched = true;
          arr = [];
          console.log(obj,input/obj,'awanish');
          myArr = Array(input/obj).fill(obj);
          return myArr;
        } else {
        	return myArr;
        }
     } else {
     return myArr
     }
   }
  );
